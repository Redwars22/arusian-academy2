# 游 Phrasebook

Here you can find ways to say the most basic words and expressions in Arusian!
You will learn how to great people and ask/reply to common questions!

import { FontRenderer } from "../../src/components/FontRenderer"

export function RenderSemlek(props) {
  return <FontRenderer size="22px" weight="900" rtl color="var(--ifm-color-primary, #794cbd)" text={props.phrase} />
}

export function Audio(props){

    const getURL = (file) => {
        const BASE = "https://thearusianlanguage.vercel.app/audio/";

        return `${BASE}${file}`
    }

    return <>
            <div>
              <audio controls style={{width: '100px'}}>
                  <source src={getURL(props.file)} type="audio/mpeg"/>
              </audio>
            </div>
            <span>[{props.ipa}]</span>
        </>
}
