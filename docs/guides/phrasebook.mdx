# 📒 Phrasebook

Here you can find ways to say the most basic words and expressions in Arusian!
You will learn how to great people and ask/reply to common questions!

(Audio files will be uploaded soon!)

import { FontRenderer } from "../../src/components/FontRenderer"

export function RenderSemlek(props) {
  return <FontRenderer size="30px" weight="200" rtl color="var(--ifm-color-primary, #794cbd)" text={props.phrase} />
}

export function Audio(props){
    return <>
            <audio controls>
                <source src={props.file} type="audio/mpeg"/>
            </audio>
            <span>[{props.ipa}]</span>
        </>
}
